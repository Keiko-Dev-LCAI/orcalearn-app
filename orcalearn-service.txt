# To install/reinstall the OrcaLearn systemd service, run these commands:
# SECURITY: set LIGHTCHAIN_PRIVATE_KEY in the EnvironmentFile below — never commit the real key.

sudo mkdir -p /etc/orcalearn
sudo tee /etc/orcalearn/dapp-wallet.env > /dev/null << 'EOF'
LIGHTCHAIN_PRIVATE_KEY=PASTE_DAPP_KEY_HERE
EOF
sudo chmod 600 /etc/orcalearn/dapp-wallet.env

sudo tee /etc/systemd/system/orcalearn-server.service > /dev/null << 'EOF'
[Unit]
Description=OrcaLearn Homeschool Backend
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
User=keiko
WorkingDirectory=/home/keiko
EnvironmentFile=/etc/orcalearn/dapp-wallet.env
ExecStart=/usr/bin/python3 /home/keiko/Desktop/orcalearn-server.py
Restart=always
RestartSec=10
StartLimitIntervalSec=60
StartLimitBurst=5

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable orcalearn-server
sudo systemctl start orcalearn-server
sudo systemctl status orcalearn-server
